Introduction

In this analysis, I compare EBICA with a Rademacher prior, EBICA with a generalized Rademacher prior (where the prior probability \(\pi\) is estimated by maximizing the marginal likelihood), and fastICA, on the whitened data with and without an intercept in the 9 unbalanced group examples.

Without an intercept, none of the three approaches appears to capture one true source; for EBICA generalized Rademacher, the estimated prior probability also seems far off from the true 0.2/0.8 probabilities.

With the intercept added, EBICA generalized Rademacher always picks up the trivial solution, with \(\hat{\pi}=1/0\), which is not ideal; I wonder if it makes sense to put a prior on \(\pi\) to shrink it toward \(0.5\), say \(\text{Beta}(5, 5)\)? EBICA Rademacher, on the other hand, is very similar to fastICA as expected, but fastICA seems to capture the trivial solution slightly more often, probably due to the Newton “acceleration”.

flash_x_rademacher <- function(X, S_init, ebnm_fn = ebnm_normal, ebnm_source = ebnm_rademacher, S = NULL, var_type = 1, backfit = TRUE, maxiter = 30) {
   if (is.vector(S_init)) S_init <- matrix(S_init, ncol = 1)
  A_init = t(solve(crossprod(S_init), crossprod(S_init, t(X))))
  if (!is.null(S)) {
    var_type = NULL
  }
  lambda_path = c()
  elbo_path <- c()
  resid_path <- c()
  z_prior_var_path <- c()

  res <- flash_init(X, var_type = var_type, S = S) |>
      flash_factors_init(
        init = list(A_init, S_init),
        ebnm_fn = list(ebnm_fn, ebnm_source)
  )
  lambda_path[1] = t(S_init) %*% t(X) %*% X %*% S_init
  elbo_path[1] = res$elbo
  resid_path[1] = res$residuals_sd
  z_prior_var_path[1] = res$L_ghat[[1]]$sd
  for (i in 1:maxiter) {
    res <- res |>
      flash_backfit(verbose = 0, extrapolate = TRUE, maxiter = 1)
    lambda_path[i+1] = t(res$F_pm) %*% t(X) %*% X %*% res$F_pm
    elbo_path[i+1] = res$elbo
    resid_path[i+1] = res$residuals_sd
    z_prior_var_path[i+1] = res$L_ghat[[1]]$sd
  }
  return(list(res=res, lambda_path = lambda_path, 
              elbo_path = elbo_path, resid_path = resid_path,
              z_prior_var_path = z_prior_var_path))
}
